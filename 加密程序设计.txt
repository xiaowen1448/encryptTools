1. 机器码生成（Machine Code）—— 唯一硬件指纹使用 WMI 获取多硬件信息 + SHA256 哈希，保证同一台机器每次生成一致。csharp

using System;
using System.Management;
using System.Security.Cryptography;
using System.Text;

public class MachineCodeHelper
{
    public static string GetMachineCode()
    {
        string cpu = GetHardwareInfo("Win32_Processor", "ProcessorId");
        string disk = GetHardwareInfo("Win32_DiskDrive", "SerialNumber");
        string bios = GetHardwareInfo("Win32_BIOS", "SerialNumber");
        string board = GetHardwareInfo("Win32_BaseBoard", "SerialNumber");

        string raw = cpu + disk + bios + board;
        using var sha256 = SHA256.Create();
        byte[] hash = sha256.ComputeHash(Encoding.UTF8.GetBytes(raw));
        return BitConverter.ToString(hash).Replace("-", "").ToLower(); // 64位十六进制
    }

    private static string GetHardwareInfo(string wmiClass, string property)
    {
        try
        {
            using var searcher = new ManagementObjectSearcher(wmiClass);
            foreach (ManagementObject obj in searcher.Get())
            {
                return obj[property]?.ToString() ?? "";
            }
        }
        catch { }
        return "";
    }
}

using System;
using System.IO;
using System.Security.Cryptography;
using System.Text;
using System.Text.Json;

public class LicenseInfo
{
    public string MachineCode { get; set; } = "";
    public DateTime ExpireDate { get; set; }
    public string Features { get; set; } = "full"; // 可扩展功能模块
}

public class LicenseHelper
{
    private static readonly byte[] Key = Encoding.UTF8.GetBytes("Your32ByteSecretKeyHere12345678"); // 32字节密钥，自己改
    private static readonly byte[] IV = Encoding.UTF8.GetBytes("Your16ByteIVHere"); // 16字节IV

    // 生成加密 LIC 文件
    public static void GenerateLic(string machineCode, DateTime expire, string outputPath)
    {
        var lic = new LicenseInfo { MachineCode = machineCode, ExpireDate = expire };
        string json = JsonSerializer.Serialize(lic);

        using var aes = Aes.Create();
        aes.Key = Key;
        aes.IV = IV;

        using var encryptor = aes.CreateEncryptor();
        byte[] encrypted = encryptor.TransformFinalBlock(Encoding.UTF8.GetBytes(json), 0, json.Length);

        File.WriteAllBytes(outputPath, encrypted);
        Console.WriteLine($"LIC 生成成功：{outputPath}");
    }

    // 读取并解密 LIC
    public static LicenseInfo? LoadLic(string licPath)
    {
        if (!File.Exists(licPath)) return null;
        byte[] data = File.ReadAllBytes(licPath);

        using var aes = Aes.Create();
        aes.Key = Key;
        aes.IV = IV;

        using var decryptor = aes.CreateDecryptor();
        byte[] decrypted = decryptor.TransformFinalBlock(data, 0, data.Length);

        string json = Encoding.UTF8.GetString(decrypted);
        return JsonSerializer.Deserialize<LicenseInfo>(json);
    }
}
