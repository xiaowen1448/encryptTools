🔧 加密代码（生成 payload.bin）
using System.Security.Cryptography;

byte[] raw = File.ReadAllBytes("CoreApp.dll");

byte[] key = Encoding.UTF8.GetBytes("1234567890123456"); // 16位
byte[] iv  = Encoding.UTF8.GetBytes("6543210987654321");

byte[] encrypted;

using (Aes aes = Aes.Create())
{
    aes.Key = key;
    aes.IV = iv;

    using var ms = new MemoryStream();
    using var cs = new CryptoStream(ms, aes.CreateEncryptor(), CryptoStreamMode.Write);

    cs.Write(raw, 0, raw.Length);
    cs.Close();

    encrypted = ms.ToArray();
}

File.WriteAllBytes("payload.bin", encrypted);

四、主程序实现（核心）
✅ 1. AES解密函数
byte[] Decrypt(byte[] encrypted)
{
    byte[] key = Encoding.UTF8.GetBytes("1234567890123456");
    byte[] iv  = Encoding.UTF8.GetBytes("6543210987654321");

    using var aes = Aes.Create();
    aes.Key = key;
    aes.IV = iv;

    using var ms = new MemoryStream();
    using var cs = new CryptoStream(ms, aes.CreateDecryptor(), CryptoStreamMode.Write);

    cs.Write(encrypted, 0, encrypted.Length);
    cs.Close();

    return ms.ToArray();
}
✅ 2. License 校验（必须先执行）
bool CheckLicense()
{
    string path = Path.Combine(
        Environment.GetFolderPath(Environment.SpecialFolder.ApplicationData),
        "YourApp",
        "license.dat"
    );

    if (!File.Exists(path)) return false;

    var json = File.ReadAllText(path);
    var lic = JsonSerializer.Deserialize<License>(json);

    string raw = $"{lic.machine}|{lic.expire}|{lic.type}|salt";

    if (!VerifySign(raw, lic.sign))
        return false;

    if (lic.machine != MachineCode.Get())
        return false;

    if (DateTime.Now > DateTime.Parse(lic.expire))
        return false;

    return true;
}
✅ 3. 内存加载执行（核心 ⭐）
👉 如果是 DLL
using System.Reflection;

void RunPayload()
{
    byte[] encrypted = Properties.Resources.payload;

    byte[] raw = Decrypt(encrypted);

    Assembly asm = Assembly.Load(raw);

    var type = asm.GetType("CoreApp.Program");
    var method = type.GetMethod("Main");

    method.Invoke(null, new object[] { new string[0] });

    Array.Clear(raw, 0, raw.Length); // 清内存
}

五、导入 License（UI）
void ImportLicense()
{
    OpenFileDialog dlg = new OpenFileDialog();
    dlg.Filter = "License|*.lic";

    if (dlg.ShowDialog() == DialogResult.OK)
    {
        string content = File.ReadAllText(dlg.FileName);

        if (!Validate(content))
        {
            MessageBox.Show("无效授权");
            return;
        }

        string path = Path.Combine(
            Environment.GetFolderPath(Environment.SpecialFolder.ApplicationData),
            "YourApp",
            "license.dat"
        );

        Directory.CreateDirectory(Path.GetDirectoryName(path));
        File.WriteAllText(path, content);

        MessageBox.Show("授权成功");
    }
}
